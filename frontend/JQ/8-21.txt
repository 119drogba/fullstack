js 이벤트핸들러함수 속성
객체나엘리먼트명.onblur = function(){ ... };

jquery이벤트핸들러 함수
blur(function(){ ... });

**이벤트 종류 : 이벤트함수의 매개 변수에서 이벤트 type을
넣는 매개변수 값을 의미(문자열로 기술)
이벤트 종류: "blur"

js
객체나엘리먼트명.addEventListener('이벤트종류',function(){...});

jquery
$(객체명).on('이벤트종류',function(){ ... });
$(객체명).bind('이벤트종류',function(){ ... });

jquery ui 라이브러리 : 전반적인 구조와 스타일, 기능등이 구현되어진
완전한 프레임워크를 제공해주는 자바스크립트 기반의 라이브러리를 말함.
필요한 만큼 라이브러리를 추가해서 사용할 수 있다.
